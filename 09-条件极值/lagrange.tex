\begin{frame}
    \frametitle{拉格朗日乘数法}

    \begin{block}{目标}
        在约束条件
        \begin{equation}
            F(x,y,u,v)=0, G(x,y,u,v)=0
            \label{eq:constraint}
        \end{equation}
        下，求解$f(x,y,u,v)$的极值。
    \end{block}
    \alert{定理18.3}
    若$f(x,y,u,v),F(x,y,u,v),G(x,y,u,v)$在点$P_0(x_0,y_0,u_0,v_0)$的某邻域有连续的偏导数，且函数$f$在约束条件(\ref{eq:constraint})下在$P_0$点取极值，矩阵
    $$
    \left.
    \begin{pmatrix}
        F_x & F_y & F_u & F_v \\
        G_x & G_y & G_u & G_v
    \end{pmatrix}
    \right|_{p_0}
    $$
    的秩为2，则存在唯一确定的$\lambda_1,\lambda_2$，使得$L_x(P_0)=L_y(P_0)=L_u(P_0)=L_v(P_0)=0$，$F(P_0)=G(P_0)=0$。

\end{frame}
